x = np.linspace(0, 10, 1000)  # Create 1000 points between [0, 10]
y = np.sin(x)
z = np.cos(x)

plt.plot(x, y, label="sin(x)")
plt.plot(x, z, label="cos(x)")

plt.grid()
plt.title("Two functions")
plt.xlabel("x-values")
plt.ylabel("y-values")
plt.legend()
plt.show()
