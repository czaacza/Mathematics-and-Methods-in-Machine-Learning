income_average = summary.loc['mean']['Income'] # store the mean value of the income in variable income_average

for index, row in df.iterrows(): # Iterating through the rows in the original data frame
    if row["Income"] is None: # Check if the income value is null
        print(f'Person {index}: no value.') # Print no value if the income value is null
    elif row["Income"] < income_average: # Otherwise, check if the income is smaller than the income average
        print(f'Person {index}: below average income.') # Print the person's income is below average income
    elif row["Income"] > income_average: # Otherwise, check if the income is bigger than the income average
        print(f'Person {index}: above average income.') # Print the person's income is above average income
    else:
        print(f'Person {index}: as same as average income.') # Otherwise, print the person's income is above average income
